CLOCK CLK 1;
SWITCH FV 1;
SWITCH ZV 0;

DTYPE D0;
DTYPE D1;
DTYPE D2;
DTYPE D3;

D0.QBAR -> D0.DATA;
D1.QBAR -> D1.DATA;
D2.QBAR -> D2.DATA;
D3.QBAR -> D3.DATA;

ZV -> D0.SET;
ZV -> D1.SET;
ZV -> D2.SET;
ZV -> D3.SET;

ZV -> D0.CLEAR;
ZV -> D1.CLEAR;
ZV -> D2.CLEAR;
ZV -> D3.CLEAR;

CLK -> D0.CLK;

D0.QBAR -> D1.CLK;
D1.QBAR -> D2.CLK;
D2.QBAR -> D3.CLK;


MONITOR CLK D0.Q D1.Q D2.Q D3.Q;
