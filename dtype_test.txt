CLOCK CLK1 1;
SWITCH SW1 0;
SWITCH SW2 0;

DTYPE D1;
DTYPE D2;
DTYPE D3;
DTYPE D4;

SW2 -> D1.SET; 
SW2 -> D1.CLEAR; 
SW2 -> D2.SET; 
SW2 -> D2. CLEAR; 
SW2 -> D3.SET; 
SW2 -> D3. CLEAR; 
SW2 -> D4.SET; 
SW2 -> D4. CLEAR;

CLK1 -> D1.CLK;
CLK1 -> D2.CLK;
CLK1 -> D3.CLK;
CLK1 -> D4.CLK;
SW1 -> D1.DATA;
D1.Q -> D2.DATA;
D2.Q -> D3.DATA;
D3.Q -> D4.DATA;

MONITOR CLK1 D1.Q D2.Q D3.Q D4.Q;
